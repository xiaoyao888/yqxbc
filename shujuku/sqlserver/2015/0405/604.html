    --增加列
    alter table SeekWorker add BornID varchar(30) not null default ''

    --修改列
    alter table CorpInf alter column CorpBriefing varchar(4000)

    --增加一个 NOT NULL 约束
    ALTER TABLE test_tab
      ALTER COLUMN id INT NOT NULL
     go

    --设为主键
    ALTER TABLE test_tab
      ADD CONSTRAINT pk_test_tab PRIMARY KEY(id);
     go
